« Écris un debounce en TypeScript, typé correctement »

function debounce<T extends (...args: any[]) => void>(
  fn: T,
  delai = 300
): (...args: Parameters<T>) => void {
  let minuteur: number | undefined;
  return (...args) => {
    clearTimeout(minuteur);
    minuteur = setTimeout(() => fn(...args), delai);
  };
}
